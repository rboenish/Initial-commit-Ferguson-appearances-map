   311→        // Generate realistic climate data
   312→        function generateClimateData() {
   313→            const data = {};
   314→            const startYear = 1924;
   315→            const endYear = 2024;
   316→
   317→            for (const [city, config] of Object.entries(cities)) {
   318→                data[city] = [];
   319→
   320→                for (let year = startYear; year <= endYear; year++) {
   321→                    // Add realistic variation and slight warming trend
   322→                    const yearOffset = (year - 1924) / 100;
   323→                    const tempVariation = (Math.sin(year * 0.3) * 2) + (Math.random() - 0.5) * 3;
   324→                    const precipVariation = (Math.cos(year * 0.25) * 4) + (Math.random() - 0.5) * 8;
   325→
   326→                    // Slight warming trend over 100 years
   327→                    const warmingTrend = yearOffset * 2.5;
   328→
   329→                    data[city].push({
   330→                        year: year,
   331→                        temperature: Math.round((config.baseTemp + tempVariation + warmingTrend) * 10) / 10,
   332→                        precipitation: Math.round((config.basePrecip + precipVariation) * 10) / 10
   333→                    });
   334→                }
   335→            }
   336→            return data;
   337→        }
